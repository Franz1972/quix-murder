can_set_relation_bodyguard_trigger = {
	NOR = {
		has_relation_bodyguard = $CHARACTER$
		has_relation_assassin = $CHARACTER$
	}
}
can_set_relation_assassin_trigger = {
	NOR = {
		has_relation_bodyguard = $CHARACTER$
		has_relation_assassin = $CHARACTER$
	}
}
